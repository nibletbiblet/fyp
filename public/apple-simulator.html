import jwt from 'jsonwebtoken';
import fetch from 'node-fetch';

// Apple Callback Route Handler
app.post('/api/auth/apple/callback', async (req, res) => {
  const { code, id_token, user } = req.body;
  
  try {
    // 1. Verify id_token JWT (Signed by Apple Public Key)
    const decoded = jwt.decode(id_token);
    console.log('Decoded Apple user profile:', decoded);
    
    // 2. Parse initial registration info (only sent on FIRST sign-in)
    if (user) {
      const { name, email } = JSON.parse(user);
      console.log('Registering user details:', name.firstName, name.lastName, email);
    }
    
    // 3. Establish session / generate local credentials JWT
    res.json({ success: true, message: 'Successfully authenticated!', user: decoded });
  } catch (err) {
    res.status(400).json({ success: false, error: err.message });
  }
});
